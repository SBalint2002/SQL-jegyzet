------------------------------------------------------------------------------------------------------------------------------------------------------
G-vel kezdődők fizetése
------------------------------------------------------------------------------------------------------------------------------------------------------
SET SERVEROUTPUT ON

DECLARE

    CURSOR emp_cursor IS
    SELECT employee_id, last_name, salary FROM employees
    WHERE last_name LIKE 'G%';
    V_osszeg NUMBER NOT NULL := 0;

BEGIN
    FOR emp_record IN emp_cursor
        LOOP
            V_osszeg := V_osszeg + emp_record.salary;
        END LOOP;
    DBMS_OUTPUT.PUT_LINE('G-vel kezdődők fizetése: ' ||V_osszeg);
END;
/


------------------------------------------------------------------------------------------------------------------------------------------------------
Procedúra készítése:
------------------------------------------------------------------------------------------------------------------------------------------------------
CREATE OR REPLACE PROCEDURE LIST_SALARIES IS
    CURSOR emp_cursor IS SELECT salary, last_name, first_name FROM employees WHERE employees.last_name LIKE ('G%');
BEGIN
    FOR emp_record IN emp_cursor LOOP
        DBMS_OUTPUT.PUT_LINE(emp_record.last_name ||' ' ||emp_record.first_name || ' ' || emp_record.salary);
    END LOOP;
END;
/
BEGIN
LIST_SALARIES;
END;
/

Funkció készítése:

CREATE OR REPLACE FUNCTION FAKTOR (szam NUMBER)
RETURN number IS eredmeny NUMBER := 1;

BEGIN
    FOR i in 1..szam LOOP
    eredmeny := eredmeny * i;
    END LOOP;
    return eredmeny;
END;
/

BEGIN
DBMS_OUTPUT.PUT_LINE(FAKTOR(5));
END;
